Building a complete expense management system​

This example shows how to build an end-to-end system that processes receipt images, extracts data, and stores it in a database.

#!/usr/bin/env python3
import requests
import json
from datetime import datetime
import os

# Configuration
PARSEFLOW_API_KEY = os.getenv("PARSEFLOW_API_KEY")
PARSEFLOW_BASE = "https://jsonparser-production-f9af.up.railway.app"

# Define expense schema for consistency
EXPENSE_SCHEMA = {
  "fields": [
    {"name": "vendor_name", "type": "string", "required": True},
    {"name": "date", "type": "date", "required": True},
    {"name": "amount", "type": "number", "required": True},
    {"name": "category", "type": "string"},
    {"name": "payment_method", "type": "string"},
    {"name": "receipt_id", "type": "string"}
  ]
}

class ExpenseProcessor:
    def __init__(self, api_key):
        self.api_key = api_key
        self.session = requests.Session()
        self.session.headers.update({
            "X-API-Key": api_key,
            "X-Client-Id": "expense-management-system"
        })
    
    def process_receipt(self, file_path, receipt_id=None):
        """Process a single receipt image/PDF"""
        
        # Prepare the request
        files = {
            "file": open(file_path, "rb"),
        }
        data = {
            "preset": "extraction-focused",
            "mode": "deterministic",
            "schema_json": json.dumps(EXPENSE_SCHEMA)
        }
        
        # Send to Parseflow
        response = self.session.post(
            f"{PARSEFLOW_BASE}/v2/process",
            files=files,
            data=data
        )
        
        if response.status_code != 200:
            raise Exception(f"Parseflow error: {response.text}")
        
        result = response.json()
        
        # Extract fields
        extraction = result.get("extraction", {}).get("fields", {})
        
        # Build expense record
        expense = {
            "receipt_id": receipt_id or result.get("request_id"),
            "processed_at": datetime.now().isoformat(),
            "raw_extraction": extraction,
            "is_confident": self._check_confidence(extraction)
        }
        
        # Map extracted fields to database schema
        if "vendor_name" in extraction:
            expense["vendor"] = extraction["vendor_name"].get("value")
        if "amount" in extraction:
            expense["amount"] = extraction["amount"].get("value")
        if "date" in extraction:
            expense["date"] = extraction["date"].get("value")
        if "category" in extraction:
            expense["category"] = extraction["category"].get("value")
        
        return expense
    
    def _check_confidence(self, fields, min_confidence=0.75):
        """Verify confidence scores are acceptable"""
        for field_name, field_data in fields.items():
            confidence = field_data.get("confidence", 0)
            if confidence < min_confidence:
                return False
        return True
    
    def batch_process(self, file_paths):
        """Process multiple receipts"""
        results = []
        for file_path in file_paths:
            try:
                result = self.process_receipt(file_path)
                results.append(result)
            except Exception as e:
                results.append({
                    "file": file_path,
                    "error": str(e)
                })
        return results

# Usage
if __name__ == "__main__":
    processor = ExpenseProcessor(PARSEFLOW_API_KEY)
    
    # Process multiple receipts
    receipts = ["receipt1.jpg", "receipt2.pdf", "receipt3.jpg"]
    results = processor.batch_process(receipts)
    
    # Store in database or send downstream
    for expense in results:
        if "error" not in expense:
            print(f"Processed: {expense['vendor']} - ${expense['amount']}")
            # db.insert_expense(expense)
        else:
            print(f"Failed: {expense['file']} - {expense['error']}")

Server-side integration with webhook callbacks​

Handle large document processing asynchronously with webhook delivery.

from flask import Flask, request, jsonify
import requests
import json
import os

app = Flask(__name__)

PARSEFLOW_API_KEY = os.getenv("PARSEFLOW_API_KEY")
PARSEFLOW_BASE = "https://jsonparser-production-f9af.up.railway.app"
YOUR_WEBHOOK_URL = os.getenv("YOUR_WEBHOOK_URL")  # e.g., https://myapp.com/webhooks/parseflow

# In-memory job tracker (use a database in production)
jobs = {}

@app.route('/upload-document', methods=['POST'])
def upload_document():
    """User uploads a document; we submit async job"""
    
    if 'file' not in request.files:
        return jsonify({"error": "No file provided"}), 400
    
    file = request.files['file']
    client_id = request.form.get('client_id', 'unknown')
    
    # Submit async job to Parseflow
    files = {
        "file": (file.filename, file.stream, file.content_type)
    }
    data = {
        "preset": "rag-optimized",
        "callback_url": YOUR_WEBHOOK_URL
    }
    
    headers = {"X-API-Key": PARSEFLOW_API_KEY, "X-Client-Id": client_id}
    
    response = requests.post(
        f"{PARSEFLOW_BASE}/v2/jobs",
        files=files,
        data=data,
        headers=headers
    )
    
    if response.status_code != 200:
        return jsonify({"error": "Failed to submit job"}), 500
    
    job_data = response.json()
    job_id = job_data.get("job_id")
    
    # Track the job
    jobs[job_id] = {
        "status": "submitted",
        "filename": file.filename,
        "client_id": client_id
    }
    
    return jsonify({
        "job_id": job_id,
        "status": "processing",
        "message": "Your document is being processed. Results will be delivered via webhook."
    }), 202

@app.route('/webhooks/parseflow', methods=['POST'])
def parseflow_webhook():
    """Parseflow delivers results here when ready"""
    
    data = request.json
    job_id = data.get("job_id")
    status = data.get("status")
    result = data.get("result")
    
    # Update job status
    if job_id in jobs:
        jobs[job_id]["status"] = status
        jobs[job_id]["result"] = result
        
        if status == "completed":
            # Process the result
            handle_completed_job(job_id, result)
        elif status == "failed":
            handle_failed_job(job_id, data.get("error"))
    
    return jsonify({"status": "received"}), 200

        "text_content": parsed_result.get("text"),
        "metadata": parsed_result.get("metadata"),
        "chunks": parsed_result.get("chunks", []),
        "completed_at": datetime.now().isoformat()
    }
    
    # db.insert_parsed_document(db_record)
    
    # Notify user (send email, update UI, etc.)

    """Handle parsing failures"""
    job_info = jobs[job_id]
    print(f"Job {job_id} failed: {error}")
    notify_user_job_failed(job_info["client_id"], job_id, error)

@app.route('/jobs/<job_id>', methods=['GET'])
def get_job_status(job_id):
    """Check job status"""
    if job_id not in jobs:
        return jsonify({"error": "Job not found"}), 404
    
    return jsonify(jobs[job_id]), 200

Client-side integration for web apps​

Send documents from a web app and update UI as results come in.

// document-upload.js
class ParseflowUploader {
  constructor(apiKey, baseUrl = 'https://jsonparser-production-f9af.up.railway.app') {
    this.apiKey = apiKey;
    this.baseUrl = baseUrl;
  }

  async uploadAndProcess(file, options = {}) {
    const formData = new FormData();
    formData.append('file', file);
    formData.append('preset', options.preset || 'extraction-focused');
    formData.append('mode', options.mode || 'deterministic');
    
    if (options.schema) {
      formData.append('schema_json', JSON.stringify(options.schema));
    }

    try {
      const response = await fetch(`${this.baseUrl}/v2/process`, {
        method: 'POST',
        headers: {
          'X-API-Key': this.apiKey,
          'X-Client-Id': options.clientId || 'web-app'
        },
        body: formData
      });

      if (!response.ok) {
        throw new Error(`Upload failed: ${response.statusText}`);
      }

      const result = await response.json();
      return this.processResult(result);
    } catch (error) {
      console.error('Error:', error);
      throw error;
    }
  }

  processResult(result) {
    // Extract fields from response
    const extraction = result.extraction || {};
    const fields = extraction.fields || {};

    // Return formatted data
    return {
      requestId: result.request_id,
      confidence: extraction.average_confidence,
      fields: Object.entries(fields).reduce((acc, [key, field]) => {
        acc[key] = {
          value: field.value,
          confidence: field.confidence
        };
        return acc;
      }, {}),
      raw: result
    };
  }
}

// Usage in web app
const uploader = new ParseflowUploader('YOUR_API_KEY');

document.getElementById('upload-form').addEventListener('submit', async (e) => {
  e.preventDefault();
  
  const file = document.getElementById('file-input').files[0];
  
  try {
    showLoading();
    const result = await uploader.uploadAndProcess(file, {
      preset: 'extraction-focused',
      schema: {
        fields: [
          { name: 'invoice_number' },
          { name: 'amount' },
          { name: 'date' }
        ]
      }
    });
    
    displayResults(result);
  } catch (error) {
    showError(error.message);
  } finally {
    hideLoading();
  }
});

function displayResults(result) {
  const resultsDiv = document.getElementById('results');
  let html = '<h3>Extraction Results</h3>';
  
  Object.entries(result.fields).forEach(([key, field]) => {
    const confidence = (field.confidence * 100).toFixed(0);
    html += `
      <div class="field">
        <label>${key}</label>
        <value>${field.value} <span class="confidence">(${confidence}% confident)</span></value>
      </div>
    `;
  });
  
  resultsDiv.innerHTML = html;
  resultsDiv.style.display = 'block';
}

Error handling and retry logic​

Production-grade error handling with exponential backoff.

import time
import random
from typing import Optional

class RetryableParseflowClient:
    def __init__(self, api_key, max_retries=3, base_delay=1):
        self.api_key = api_key
        self.max_retries = max_retries
        self.base_delay = base_delay
        self.session = requests.Session()
    
    def process_with_retry(self, file_path, **kwargs) -> Optional[dict]:
        """Process file with automatic retry on failure"""
        
        last_error = None
        
        for attempt in range(self.max_retries):
            try:
                return self._process(file_path, **kwargs)
            except requests.exceptions.Timeout:
                last_error = "Request timeout"
                print(f"Attempt {attempt + 1}: Timeout, retrying...")
            except requests.exceptions.ConnectionError:
                last_error = "Connection error"
                print(f"Attempt {attempt + 1}: Connection error, retrying...")
            except requests.exceptions.HTTPError as e:
                if e.response.status_code in [429, 503, 504]:  # Retryable errors
                    last_error = f"Server error: {e.response.status_code}"
                    print(f"Attempt {attempt + 1}: {last_error}, retrying...")
                else:
                    raise  # Non-retryable HTTP error
            
            # Exponential backoff with jitter
            if attempt < self.max_retries - 1:
                delay = self.base_delay * (2 ** attempt) + random.uniform(0, 1)
                print(f"Waiting {delay:.1f} seconds before retry...")
                time.sleep(delay)
        
        raise Exception(f"Max retries exceeded. Last error: {last_error}")
    
    def _process(self, file_path, **kwargs):
        """Actual processing call"""
        files = {"file": open(file_path, "rb")}
        headers = {"X-API-Key": self.api_key}
        
        response = self.session.post(
            "https://jsonparser-production-f9af.up.railway.app/v2/process",
            files=files,
            data=kwargs,
            headers=headers,
            timeout=30
        )
        
        response.raise_for_status()
        return response.json()

# Usage
client = RetryableParseflowClient("YOUR_API_KEY")

try:
    result = client.process_with_retry(
        "document.pdf",
        preset="extraction-focused",
        mode="deterministic"
    )
    print("Success:", result)
except Exception as e:
    print(f"Failed after retries: {e}")

Monitoring and analytics​

Track API usage, costs, and performance.

import json
from datetime import datetime
from collections import defaultdict

class ParseflowAnalytics:
    def __init__(self):
        self.requests = []
        self.stats = defaultdict(lambda: {
            "count": 0,
            "total_tokens": 0,
            "total_cost": 0,
            "avg_confidence": 0
        })
    
    def track_request(self, result):
        """Record a completed request"""
        
        request_data = {
            "timestamp": datetime.now().isoformat(),
            "request_id": result.get("request_id"),
            "preset": result.get("preset", {}).get("name"),
            "input_tokens": result.get("costs", {}).get("input_tokens", 0),
            "output_tokens": result.get("costs", {}).get("output_tokens", 0),
            "estimated_cost": result.get("costs", {}).get("estimated_provider_cost", 0),
            "average_confidence": result.get("extraction", {}).get("average_confidence", 0)
        }
        
        self.requests.append(request_data)
        self._update_stats(request_data)
    
    def _update_stats(self, request_data):
        """Update aggregated statistics"""
        preset = request_data["preset"] or "unknown"
        
        stats = self.stats[preset]
        stats["count"] += 1
        stats["total_tokens"] += request_data["input_tokens"] + request_data["output_tokens"]
        stats["total_cost"] += request_data["estimated_cost"]
        stats["avg_confidence"] = (
            (stats["avg_confidence"] * (stats["count"] - 1) + request_data["average_confidence"]) /
            stats["count"]
        )
    
    def get_summary(self):
        """Get usage summary"""
        total_requests = sum(s["count"] for s in self.stats.values())
        total_cost = sum(s["total_cost"] for s in self.stats.values())
        
        summary = {
            "total_requests": total_requests,
            "total_cost": round(total_cost, 6),
            "by_preset": dict(self.stats)
        }
        
        return summary
    
    def export_to_csv(self, filename):
        """Export request history to CSV for analysis"""
        import csv
        
        with open(filename, 'w', newline='') as f:
            writer = csv.DictWriter(f, fieldnames=self.requests[0].keys())
            writer.writeheader()
            writer.writerows(self.requests)
